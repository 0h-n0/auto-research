"""tinker/train.py — vision-classification domain (CIFAR-10) — auto-research v0.11.0+.

Single-file CNN trainer for autonomous tinker mode. The agent edits this file freely
to maximize val_acc on the held-out CIFAR-10 test set under a fixed wall-clock budget.

Hard rules (mirrored from program.md):
  - Do NOT modify prepare.py, data/, or the val split.
  - Do NOT use pretrained weights or torchvision pretrained models.
  - Always write tinker/result.json with at least:
      {"primary_metric": <float|null>, "metric_name": "val_acc", "direction": "max",
       "wall_time_s": <float>, "n_iters": <int>, "diverged": <bool>,
       "domain": "vision-classification"}

Acknowledgement:
  Inspired by karpathy/autoresearch (https://github.com/karpathy/autoresearch, MIT, 2026).
  Self-authored minimal CNN for the auto-research plugin (v0.11.0+).
"""

from __future__ import annotations

import json
import math
import os
import time
from dataclasses import dataclass, asdict
from pathlib import Path

import torch
import torch.nn as nn
import torch.nn.functional as F


@dataclass
class Config:
    # Architecture (agent freely changes)
    n_filters: int = 64
    depth: int = 3            # number of conv blocks
    fc_hidden: int = 256
    dropout: float = 0.0
    use_bn: bool = True

    # Optimization
    lr: float = 1e-3
    weight_decay: float = 1e-4
    grad_clip: float = 1.0
    warmup_iters: int = 100
    min_lr_ratio: float = 0.1

    # Batch
    batch_size: int = 128
    eval_batch_size: int = 256

    # Augmentation knobs (agent can flip these on/off)
    aug_flip: bool = True
    aug_crop_pad: int = 4

    seed: int = 42
    budget_seconds: int = int(os.environ.get("TINKER_BUDGET_SECONDS", 300))


def load_split(name: str):
    p = Path(__file__).parent / "data" / f"{name}.pt"
    blob = torch.load(p, weights_only=False)
    return blob["x"], blob["y"]


def augment(x: torch.Tensor, cfg: Config, gen: torch.Generator) -> torch.Tensor:
    """Simple online augmentation (agent may rewrite or remove)."""
    if cfg.aug_flip:
        flip_mask = torch.rand(x.size(0), generator=gen) < 0.5
        x[flip_mask] = torch.flip(x[flip_mask], dims=[-1])
    if cfg.aug_crop_pad > 0:
        pad = cfg.aug_crop_pad
        x = F.pad(x, (pad, pad, pad, pad), mode="reflect")
        h, w = x.shape[-2], x.shape[-1]
        starts_y = torch.randint(0, 2 * pad + 1, (x.size(0),), generator=gen)
        starts_x = torch.randint(0, 2 * pad + 1, (x.size(0),), generator=gen)
        cropped = torch.empty_like(x[:, :, : h - 2 * pad, : w - 2 * pad])
        for i in range(x.size(0)):
            sy = int(starts_y[i].item())
            sx = int(starts_x[i].item())
            cropped[i] = x[i, :, sy : sy + 32, sx : sx + 32]
        x = cropped
    return x


class ConvBlock(nn.Module):
    def __init__(self, c_in: int, c_out: int, use_bn: bool, dropout: float):
        super().__init__()
        self.conv1 = nn.Conv2d(c_in, c_out, kernel_size=3, padding=1, bias=not use_bn)
        self.bn1 = nn.BatchNorm2d(c_out) if use_bn else nn.Identity()
        self.conv2 = nn.Conv2d(c_out, c_out, kernel_size=3, padding=1, bias=not use_bn)
        self.bn2 = nn.BatchNorm2d(c_out) if use_bn else nn.Identity()
        self.drop = nn.Dropout2d(dropout) if dropout > 0 else nn.Identity()

    def forward(self, x):
        x = F.relu(self.bn1(self.conv1(x)))
        x = F.relu(self.bn2(self.conv2(x)))
        x = self.drop(x)
        return F.max_pool2d(x, 2)


class CNN(nn.Module):
    def __init__(self, cfg: Config, n_classes: int = 10):
        super().__init__()
        c = cfg.n_filters
        blocks = []
        c_prev = 3
        for i in range(cfg.depth):
            blocks.append(ConvBlock(c_prev, c, cfg.use_bn, cfg.dropout))
            c_prev = c
            c = c * 2
        self.features = nn.Sequential(*blocks)
        # Calculate flatten size after `depth` poolings of 32×32 input
        spatial = 32
        for _ in range(cfg.depth):
            spatial = spatial // 2
        flat = c_prev * spatial * spatial
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(flat, cfg.fc_hidden),
            nn.ReLU(inplace=True),
            nn.Dropout(cfg.dropout),
            nn.Linear(cfg.fc_hidden, n_classes),
        )

    def forward(self, x):
        return self.classifier(self.features(x))


def lr_at(step: int, total_steps: int, cfg: Config) -> float:
    if step < cfg.warmup_iters:
        return cfg.lr * (step + 1) / cfg.warmup_iters
    if total_steps <= cfg.warmup_iters:
        return cfg.lr * cfg.min_lr_ratio
    progress = (step - cfg.warmup_iters) / max(1, total_steps - cfg.warmup_iters)
    return cfg.lr * (cfg.min_lr_ratio + (1 - cfg.min_lr_ratio) * 0.5 * (1 + math.cos(math.pi * progress)))


@torch.no_grad()
def evaluate(model: CNN, x: torch.Tensor, y: torch.Tensor, batch_size: int, device: str) -> float:
    model.eval()
    n = x.size(0)
    correct = 0
    for i in range(0, n, batch_size):
        xb = x[i : i + batch_size].to(device, non_blocking=True)
        yb = y[i : i + batch_size].to(device, non_blocking=True)
        logits = model(xb)
        correct += int((logits.argmax(dim=-1) == yb).sum().item())
    model.train()
    return correct / n


def main():
    cfg = Config()
    torch.manual_seed(cfg.seed)
    device = "cuda" if torch.cuda.is_available() else ("mps" if torch.backends.mps.is_available() else "cpu")

    train_x_cpu, train_y_cpu = load_split("train")
    val_x_cpu, val_y_cpu = load_split("val")

    model = CNN(cfg).to(device)
    optim = torch.optim.AdamW(model.parameters(), lr=cfg.lr, weight_decay=cfg.weight_decay)

    n_params = sum(p.numel() for p in model.parameters())
    print(f"# device={device} params={n_params/1e6:.2f}M depth={cfg.depth} filt={cfg.n_filters} budget={cfg.budget_seconds}s")

    t0 = time.monotonic()
    gen_aug = torch.Generator().manual_seed(cfg.seed)
    gen_perm = torch.Generator().manual_seed(cfg.seed + 1)
    step = 0
    diverged = False
    last_loss = float("inf")
    n_train = train_x_cpu.size(0)

    # Heuristic total_steps for cosine: assume ~50 step/s on mid-tier GPU
    total_steps_est = max(50, int(cfg.budget_seconds * 50))

    while True:
        if time.monotonic() - t0 >= cfg.budget_seconds * 0.85:  # leave 15% for eval
            break
        # Sample a random batch
        idx = torch.randint(0, n_train, (cfg.batch_size,), generator=gen_perm)
        xb = train_x_cpu[idx].clone()
        yb = train_y_cpu[idx]
        xb = augment(xb, cfg, gen_aug)
        xb = xb.to(device, non_blocking=True)
        yb = yb.to(device, non_blocking=True)

        lr = lr_at(step, total_steps_est, cfg)
        for g in optim.param_groups:
            g["lr"] = lr

        optim.zero_grad(set_to_none=True)
        logits = model(xb)
        loss = F.cross_entropy(logits, yb)
        if math.isnan(loss.item()) or math.isinf(loss.item()):
            diverged = True
            break
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), cfg.grad_clip)
        optim.step()

        last_loss = float(loss.item())
        step += 1
        if step % 100 == 0:
            print(f"step {step}  loss {last_loss:.4f}  lr {lr:.2e}  t {time.monotonic()-t0:.1f}s")

    val_acc: float | None
    try:
        val_acc = evaluate(model, val_x_cpu, val_y_cpu, cfg.eval_batch_size, device) if not diverged else None
    except Exception as e:
        print(f"# eval failed: {type(e).__name__}: {e}")
        val_acc = None

    wall = time.monotonic() - t0
    result = {
        "schema_version": 1,
        "domain": "vision-classification",
        "primary_metric": val_acc,
        "metric_name": "val_acc",
        "direction": "max",
        "wall_time_s": wall,
        "n_iters": step,
        "diverged": diverged,
        "last_train_loss": last_loss,
        "params_M": round(n_params / 1e6, 3),
        "device": device,
        "config": asdict(cfg),
    }
    Path("tinker/result.json").parent.mkdir(parents=True, exist_ok=True)
    Path("tinker/result.json").write_text(json.dumps(result, indent=2))
    print(f"# done val_acc={val_acc} steps={step} wall={wall:.1f}s diverged={diverged}")


if __name__ == "__main__":
    main()
