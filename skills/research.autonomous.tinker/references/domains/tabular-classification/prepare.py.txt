"""tinker/prepare.py — tabular-classification domain — auto-research v0.11.0+.

One-time data prep for sklearn's breast_cancer dataset:
  - Loads sklearn breast_cancer (569 samples, 30 features, 2 classes)
  - Stratified 80/20 train/val split with FIXED seed
  - Saves train.npz and val.npz under tinker/data/

Hard rules (agent must NOT modify this file):
  - Train/val split is FIXED here. Modifying it is data leakage.
  - Feature standardization fits on train only, applied to val.

Acknowledgement:
  Inspired by karpathy/autoresearch (https://github.com/karpathy/autoresearch, MIT, 2026).
"""

from __future__ import annotations

import json
from pathlib import Path

import numpy as np
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split


DOMAIN = "tabular-classification"
SPLIT_SEED = 12345  # FROZEN. Do not change.
VAL_FRACTION = 0.20


def main():
    data_dir = Path(__file__).parent / "data"
    data_dir.mkdir(parents=True, exist_ok=True)

    X, y = load_breast_cancer(return_X_y=True)
    X = X.astype(np.float32)
    y = y.astype(np.int64)

    X_tr, X_va, y_tr, y_va = train_test_split(
        X, y,
        test_size=VAL_FRACTION,
        random_state=SPLIT_SEED,
        stratify=y,
    )

    # Standardize using train statistics only (fit on train, apply to val)
    mean = X_tr.mean(axis=0)
    std = X_tr.std(axis=0) + 1e-8
    X_tr = (X_tr - mean) / std
    X_va = (X_va - mean) / std

    np.savez(data_dir / "train.npz", x=X_tr, y=y_tr)
    np.savez(data_dir / "val.npz", x=X_va, y=y_va)

    manifest = {
        "schema_version": 1,
        "domain": DOMAIN,
        "dataset": "sklearn.datasets.load_breast_cancer",
        "n_features": int(X.shape[1]),
        "n_classes": 2,
        "n_train": int(X_tr.shape[0]),
        "n_val": int(X_va.shape[0]),
        "split_seed": SPLIT_SEED,
        "val_fraction": VAL_FRACTION,
        "feature_mean": mean.tolist(),
        "feature_std": std.tolist(),
        "license": "BSD (sklearn-bundled)",
    }
    (data_dir / "manifest.json").write_text(json.dumps(manifest, indent=2))

    print(f"# wrote train: {X_tr.shape}, val: {X_va.shape}")
    print(f"# manifest: {data_dir / 'manifest.json'}")


if __name__ == "__main__":
    main()
