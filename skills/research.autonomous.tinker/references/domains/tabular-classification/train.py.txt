"""tinker/train.py — tabular-classification domain — auto-research v0.11.0+.

Single-file MLP trainer for autonomous tinker mode. The agent edits this file freely
to maximize val_acc on the held-out breast_cancer val split under a fixed wall-clock budget.

Hard rules:
  - Do NOT modify prepare.py or data/.
  - No AutoML / TabPFN / sklearn pretrained models.
  - Always write tinker/result.json with at least:
      {"primary_metric": <float|null>, "metric_name": "val_acc", "direction": "max",
       "wall_time_s": <float>, "n_iters": <int>, "diverged": <bool>,
       "domain": "tabular-classification"}

Acknowledgement:
  Inspired by karpathy/autoresearch (https://github.com/karpathy/autoresearch, MIT, 2026).
  Self-authored MLP for the auto-research plugin (v0.11.0+).
"""

from __future__ import annotations

import json
import math
import os
import time
from dataclasses import dataclass, asdict
from pathlib import Path

import numpy as np
import torch
import torch.nn as nn
import torch.nn.functional as F


@dataclass
class Config:
    # Architecture
    hidden: int = 64
    depth: int = 2          # number of hidden linear layers
    dropout: float = 0.0
    activation: str = "relu"  # relu | gelu | tanh

    # Optimization
    lr: float = 3e-3
    weight_decay: float = 1e-3
    grad_clip: float = 1.0
    warmup_iters: int = 50
    min_lr_ratio: float = 0.1

    # Batch
    batch_size: int = 32
    eval_batch_size: int = 64

    seed: int = 42
    budget_seconds: int = int(os.environ.get("TINKER_BUDGET_SECONDS", 300))


def load_split(name: str):
    p = Path(__file__).parent / "data" / f"{name}.npz"
    blob = np.load(p)
    return torch.from_numpy(blob["x"]), torch.from_numpy(blob["y"])


def manifest() -> dict:
    return json.loads((Path(__file__).parent / "data" / "manifest.json").read_text())


class MLP(nn.Module):
    def __init__(self, n_features: int, n_classes: int, cfg: Config):
        super().__init__()
        layers: list[nn.Module] = []
        d = n_features
        for _ in range(cfg.depth):
            layers.append(nn.Linear(d, cfg.hidden))
            layers.append(_get_activation(cfg.activation))
            if cfg.dropout > 0:
                layers.append(nn.Dropout(cfg.dropout))
            d = cfg.hidden
        layers.append(nn.Linear(d, n_classes))
        self.net = nn.Sequential(*layers)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return self.net(x)


def _get_activation(name: str) -> nn.Module:
    if name == "relu":
        return nn.ReLU()
    if name == "gelu":
        return nn.GELU()
    if name == "tanh":
        return nn.Tanh()
    raise ValueError(f"unknown activation: {name}")


def lr_at(step: int, total_steps: int, cfg: Config) -> float:
    if step < cfg.warmup_iters:
        return cfg.lr * (step + 1) / cfg.warmup_iters
    if total_steps <= cfg.warmup_iters:
        return cfg.lr * cfg.min_lr_ratio
    progress = (step - cfg.warmup_iters) / max(1, total_steps - cfg.warmup_iters)
    return cfg.lr * (cfg.min_lr_ratio + (1 - cfg.min_lr_ratio) * 0.5 * (1 + math.cos(math.pi * progress)))


@torch.no_grad()
def evaluate(model: MLP, x: torch.Tensor, y: torch.Tensor, batch_size: int, device: str) -> float:
    model.eval()
    n = x.size(0)
    correct = 0
    for i in range(0, n, batch_size):
        xb = x[i : i + batch_size].to(device)
        yb = y[i : i + batch_size].to(device)
        logits = model(xb)
        correct += int((logits.argmax(dim=-1) == yb).sum().item())
    model.train()
    return correct / n


def main():
    cfg = Config()
    torch.manual_seed(cfg.seed)
    np.random.seed(cfg.seed)
    device = "cuda" if torch.cuda.is_available() else ("mps" if torch.backends.mps.is_available() else "cpu")

    train_x, train_y = load_split("train")
    val_x, val_y = load_split("val")
    m = manifest()
    n_features = int(m["n_features"])
    n_classes = int(m["n_classes"])

    model = MLP(n_features, n_classes, cfg).to(device)
    optim = torch.optim.AdamW(model.parameters(), lr=cfg.lr, weight_decay=cfg.weight_decay)

    n_params = sum(p.numel() for p in model.parameters())
    print(f"# device={device} params={n_params/1e3:.1f}k hidden={cfg.hidden} depth={cfg.depth} budget={cfg.budget_seconds}s")

    t0 = time.monotonic()
    gen = torch.Generator().manual_seed(cfg.seed)
    step = 0
    diverged = False
    last_loss = float("inf")
    n_train = train_x.size(0)

    total_steps_est = max(50, int(cfg.budget_seconds * 200))  # MLP is fast, many steps possible

    while True:
        if time.monotonic() - t0 >= cfg.budget_seconds * 0.9:
            break
        idx = torch.randint(0, n_train, (cfg.batch_size,), generator=gen)
        xb = train_x[idx].to(device)
        yb = train_y[idx].to(device)

        lr = lr_at(step, total_steps_est, cfg)
        for g in optim.param_groups:
            g["lr"] = lr

        optim.zero_grad(set_to_none=True)
        logits = model(xb)
        loss = F.cross_entropy(logits, yb)
        if math.isnan(loss.item()) or math.isinf(loss.item()):
            diverged = True
            break
        loss.backward()
        torch.nn.utils.clip_grad_norm_(model.parameters(), cfg.grad_clip)
        optim.step()

        last_loss = float(loss.item())
        step += 1
        if step % 500 == 0:
            print(f"step {step}  loss {last_loss:.4f}  lr {lr:.2e}  t {time.monotonic()-t0:.1f}s")

    val_acc: float | None
    try:
        val_acc = evaluate(model, val_x, val_y, cfg.eval_batch_size, device) if not diverged else None
    except Exception as e:
        print(f"# eval failed: {type(e).__name__}: {e}")
        val_acc = None

    wall = time.monotonic() - t0
    result = {
        "schema_version": 1,
        "domain": "tabular-classification",
        "primary_metric": val_acc,
        "metric_name": "val_acc",
        "direction": "max",
        "wall_time_s": wall,
        "n_iters": step,
        "diverged": diverged,
        "last_train_loss": last_loss,
        "params_k": round(n_params / 1e3, 2),
        "device": device,
        "config": asdict(cfg),
    }
    Path("tinker/result.json").parent.mkdir(parents=True, exist_ok=True)
    Path("tinker/result.json").write_text(json.dumps(result, indent=2))
    print(f"# done val_acc={val_acc} steps={step} wall={wall:.1f}s diverged={diverged}")


if __name__ == "__main__":
    main()
