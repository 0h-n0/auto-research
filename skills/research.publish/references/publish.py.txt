"""scripts/publish.py — research.publish skill backend.

このファイルは reference テンプレ。実装時に rename して使う:
    cp skills/research.publish/references/publish.py.txt \\
       .research/<slug>/code/scripts/publish.py

Usage:
    python publish.py <slug> [--target hf-hub|zenodo|both] [--bundle <path>] [--draft|--no-draft]

Env:
    HF_TOKEN, HF_USERNAME (optional)
    ZENODO_ACCESS_TOKEN, ZENODO_SANDBOX=1 (optional, sandbox.zenodo.org)
"""

from __future__ import annotations

import argparse
import json
import os
import re
import subprocess
import sys
import tarfile
import tempfile
import time
from datetime import datetime, timezone
from pathlib import Path


def latest_bundle(project_dir: Path) -> Path | None:
    """Find the most recent <slug>_export_*.tar.gz next to or inside the project."""
    candidates = list(project_dir.glob(f"{project_dir.name}_export_*.tar.gz"))
    candidates += list(Path.cwd().glob(f"{project_dir.name}_export_*.tar.gz"))
    if not candidates:
        return None
    return max(candidates, key=lambda p: p.stat().st_mtime)


def hf_upload(slug: str, bundle: Path, draft: bool) -> dict | None:
    """Upload bundle to HF Hub Datasets. Returns publication info or None on skip."""
    token = os.environ.get("HF_TOKEN")
    if not token:
        print(
            "[publish] HF_TOKEN not set; skipping HF Hub upload "
            "(set https://huggingface.co/settings/tokens)",
            file=sys.stderr,
        )
        return None
    try:
        from huggingface_hub import HfApi, create_repo
    except ImportError:
        print(
            "[publish] huggingface_hub not installed; "
            "run `uv sync --extra publish` and retry",
            file=sys.stderr,
        )
        return None

    api = HfApi(token=token)
    username = os.environ.get("HF_USERNAME") or api.whoami()["name"]
    repo_id = f"{username}/auto-research-{slug}"

    # Extract bundle to temp dir for folder upload
    with tempfile.TemporaryDirectory() as td:
        with tarfile.open(bundle, "r:gz") as tar:
            tar.extractall(td)
        # Find extracted root (.research/<slug>/)
        extracted_root = next(Path(td).rglob(slug))

        create_repo(
            repo_id,
            repo_type="dataset",
            private=bool(draft),
            exist_ok=True,
            token=token,
        )
        api.upload_folder(
            folder_path=str(extracted_root),
            repo_id=repo_id,
            repo_type="dataset",
            commit_message=f"auto-research export {bundle.name}",
            token=token,
        )

    return {
        "repo_id": repo_id,
        "url": f"https://huggingface.co/datasets/{repo_id}",
        "visibility": "private (draft)" if draft else "public",
    }


def zenodo_upload(slug: str, bundle: Path, draft: bool, brief: str = "") -> dict | None:
    """Upload bundle to Zenodo. Publish if not draft. Returns DOI info or None."""
    token = os.environ.get("ZENODO_ACCESS_TOKEN")
    if not token:
        print(
            "[publish] ZENODO_ACCESS_TOKEN not set; skipping Zenodo upload "
            "(set https://zenodo.org/account/settings/applications/)",
            file=sys.stderr,
        )
        return None
    try:
        import requests  # noqa: F401
    except ImportError:
        print(
            "[publish] requests not installed; run `uv sync --extra publish` and retry",
            file=sys.stderr,
        )
        return None

    import requests as rq

    base = (
        "https://sandbox.zenodo.org/api"
        if os.environ.get("ZENODO_SANDBOX")
        else "https://zenodo.org/api"
    )
    headers = {"Authorization": f"Bearer {token}"}

    # 1. Create deposition
    for attempt in range(3):
        r = rq.post(f"{base}/deposit/depositions", json={}, headers=headers, timeout=30)
        if r.status_code == 201:
            break
        if r.status_code == 429:
            time.sleep(2 ** attempt * 5)
            continue
        r.raise_for_status()
    else:
        raise RuntimeError(f"Zenodo create_deposition failed after retries: {r.status_code}")
    dep = r.json()
    dep_id = dep["id"]
    bucket = dep["links"]["bucket"]

    # 2. Upload file
    with bundle.open("rb") as f:
        rq.put(f"{bucket}/{bundle.name}", data=f, headers=headers, timeout=600).raise_for_status()

    # 3. Metadata
    metadata = {
        "metadata": {
            "title": f"auto-research artifact: {slug}",
            "upload_type": "dataset",
            "description": brief
            or f"Research artifacts for {slug} generated by auto-research plugin.",
            "creators": [{"name": os.environ.get("ZENODO_AUTHOR", "0h-n0")}],
            "keywords": ["llm", "ml-research", "auto-research"],
        }
    }
    rq.put(
        f"{base}/deposit/depositions/{dep_id}",
        json=metadata,
        headers=headers,
        timeout=30,
    ).raise_for_status()

    # 4. Publish if not draft
    info = {
        "deposition_id": dep_id,
        "url": (
            f"https://sandbox.zenodo.org/record/{dep_id}"
            if os.environ.get("ZENODO_SANDBOX")
            else f"https://zenodo.org/record/{dep_id}"
        ),
        "status": "draft",
        "doi": None,
    }
    if not draft:
        r = rq.post(
            f"{base}/deposit/depositions/{dep_id}/actions/publish",
            headers=headers,
            timeout=30,
        )
        r.raise_for_status()
        published = r.json()
        info["doi"] = published.get("doi")
        info["status"] = "published"
        info["url"] = published.get("links", {}).get("html", info["url"])
    return info


def write_publication_md(
    slug: str,
    project_dir: Path,
    bundle: Path,
    hf: dict | None,
    zen: dict | None,
):
    out = project_dir / "PUBLICATION.md"
    git_sha = subprocess.run(
        ["git", "rev-parse", "HEAD"],
        capture_output=True,
        text=True,
        check=False,
    ).stdout.strip() or "unknown"

    lines = [
        f"# Publication Record — {slug}",
        "",
        f"Published: {datetime.now(timezone.utc).isoformat()}",
        f"git SHA: {git_sha}",
        f"Bundle: {bundle.name}",
        "",
    ]
    if hf:
        lines += [
            "## HuggingFace Hub Dataset",
            f"- URL: {hf['url']}",
            f"- Visibility: {hf['visibility']}",
            "",
        ]
    if zen:
        lines += [
            "## Zenodo Archive",
            f"- URL: {zen['url']}",
            f"- Status: {zen['status']}",
            f"- DOI: {zen.get('doi') or '(pending — draft mode)'}",
            "",
        ]
    if zen and zen.get("doi"):
        slug_clean = re.sub(r"[^a-z0-9]", "", slug)[:20]
        bibtex_key = f"{slug_clean or 'autoresearch'}{datetime.now().year}"
        lines += [
            "## Citation",
            "",
            "```bibtex",
            f"@misc{{{bibtex_key},",
            f'  author = {{{os.environ.get("ZENODO_AUTHOR", "0h-n0")}}},',
            f'  title  = {{auto-research artifact: {slug}}},',
            f"  year   = {{{datetime.now().year}}},",
            "  publisher = {Zenodo},",
            f"  doi    = {{{zen['doi']}}},",
            f'  url    = {{https://doi.org/{zen["doi"]}}}',
            "}",
            "```",
            "",
        ]
    out.write_text("\n".join(lines))
    print(f"✓ PUBLICATION.md: {out}")


def update_state(project_dir: Path, hf: dict | None, zen: dict | None):
    state_path = project_dir / "STATE.json"
    state = json.loads(state_path.read_text())
    state.setdefault("published", {})
    if hf:
        state["published"]["hf_hub"] = hf["repo_id"]
    if zen:
        if zen.get("doi"):
            state["published"]["zenodo_doi"] = zen["doi"]
        state["published"]["zenodo_url"] = zen["url"]
    state["published"]["published_at"] = datetime.now(timezone.utc).isoformat()
    state["updated_at"] = state["published"]["published_at"]
    state_path.write_text(json.dumps(state, indent=2, ensure_ascii=False))


def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("slug")
    ap.add_argument("--target", choices=["hf-hub", "zenodo", "both"], default="both")
    ap.add_argument("--bundle", type=Path, default=None)
    ap.add_argument("--draft", action="store_true", default=True)
    ap.add_argument("--no-draft", dest="draft", action="store_false")
    args = ap.parse_args()

    project_dir = Path(".research") / args.slug
    if not project_dir.exists():
        print(f"ERROR: {project_dir} not found", file=sys.stderr)
        sys.exit(1)

    bundle = args.bundle or latest_bundle(project_dir)
    if not bundle or not bundle.exists():
        print(
            "ERROR: bundle not found. Run research.export first.",
            file=sys.stderr,
        )
        sys.exit(1)

    brief_md = ""
    brief_path = project_dir / "01_BRIEF.md"
    if brief_path.exists():
        brief_md = brief_path.read_text()[:5000]  # Zenodo description size cap

    hf = zen = None
    if args.target in ("hf-hub", "both"):
        hf = hf_upload(args.slug, bundle, args.draft)
    if args.target in ("zenodo", "both"):
        zen = zenodo_upload(args.slug, bundle, args.draft, brief_md)

    if not (hf or zen):
        print("[publish] nothing was uploaded (no tokens?)", file=sys.stderr)
        sys.exit(2)

    write_publication_md(args.slug, project_dir, bundle, hf, zen)
    update_state(project_dir, hf, zen)
    print("✓ research.publish completed")


if __name__ == "__main__":
    main()
